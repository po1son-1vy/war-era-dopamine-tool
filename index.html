Case Farming Formula
# HP pool per hour
food_heal = food_pct% × max_health
effective_hp = health + (hunger × food_heal)

# Hits per hour (HP spent per attack, dodge = free hit, 0 HP cost)
dmg_per_hit = 10 × (1 − armor%)
non_dodged_hits = effective_hp / dmg_per_hit
total_attempts = non_dodged_hits / (1 − dodge%)
landed_hits = total_attempts × precision%

# Cases only proc on landed hits
cases = landed_hits × loot_chance%

# Daily: hour 1 = full HP+hunger, hours 2-24 = regen (carries over)
hp_regen = max_health × 10% / hour
hunger_regen = max_hunger × 10% / hour
cases_per_day = cases_hour1 + Σ(cases_hour2..24)

# Damage (tiebreaker only, does not affect cases)
avg_dmg = attack × (1 + crit_chance% × (crit_dmg% − 1))
dmg_per_day = avg_dmg × total_landed_hits_per_day
